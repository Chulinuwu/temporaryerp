1. ข้อกำหนดของเซิร์ฟเวอร์

รายการสเปกที่แนะนำ

หน่วยความจำ8 GB ขึ้นไป (แนะนำ 16 GB)

Web ServerNginx 1.22+ (แนะนำ) หรือ Apache 2.4

2. ตั้งค่าระบบเริ่มต้น

2.1 สร้างผู้ใช้ & SSH

sudo adduser pegasus
sudo usermod -aG sudo pegasus
sudo mkdir -p /home/pegasus/.ssh
sudo cp ~/.ssh/authorized_keys /home/pegasus/.ssh/
sudo chown -R pegasus:pegasus /home/pegasus/.ssh
sudo chmod 700 /home/pegasus/.ssh
sudo chmod 600 /home/pegasus/.ssh/authorized_keys

2.2 Firewall

sudo ufw allow OpenSSH
sudo ufw allow 80/tcp
sudo ufw allow 443/tcp
sudo ufw enable

2.3 Timezone

sudo timedatectl set-timezone Asia/Bangkok

3. ติดตั้งซอฟต์แวร์ที่จำเป็น

sudo apt update && sudo apt upgrade -y
sudo apt install -y nginx php8.2-fpm php8.2-pgsql php8.2-mbstring php8.2-gd \
    php8.2-xml php8.2-curl php8.2-zip unzip git \
    postgresql-15 postgresql-contrib-15 \
    certbot python3-certbot-nginx

4. ตั้งค่า PostgreSQL

4.1 สร้าง DB และผู้ใช้

sudo -u postgres psql <<EOF
CREATE USER pegasus_user WITH PASSWORD 'STRONG_PASSWORD_HERE';
CREATE DATABASE pegasus_erp OWNER pegasus_user ENCODING 'UTF8';
GRANT ALL PRIVILEGES ON DATABASE pegasus_erp TO pegasus_user;
EOF

4.2 ตั้งค่าการเชื่อมต่อ (pg_hba.conf)

เพิ่มใน /etc/postgresql/15/main/pg_hba.conf:

local   pegasus_erp   pegasus_user                      md5
host    pegasus_erp   pegasus_user   127.0.0.1/32       md5
sudo systemctl restart postgresql

5. วาง Application

# วางไฟล์
sudo mkdir -p /var/www/pegasus_erp
sudo chown pegasus:www-data /var/www/pegasus_erp
cd /var/www/pegasus_erp
unzip ~/pegasus_erp_source_YYYYMMDD.zip -d .

# โฟลเดอร์ที่เขียนได้
sudo mkdir -p uploads backups

6. Environment Variables / การเชื่อมต่อ DB

สร้างไฟล์ /var/www/pegasus_erp/.env:

DB_HOST=localhost
DB_PORT=5432
DB_NAME=pegasus_erp
DB_USER=pegasus_user
DB_PASS=STRONG_PASSWORD_HERE

7. เริ่มต้น Database

# กู้คืนจาก backup
PGPASSWORD='STRONG_PASSWORD_HERE' pg_restore -U pegasus_user -h localhost \
    -d pegasus_erp -c /path/to/pegasus_erp_YYYYMMDD.dump

# หรือติดตั้งใหม่
psql -U pegasus_user -d pegasus_erp -f database/schema.sql
psql -U pegasus_user -d pegasus_erp -f database/seed.sql

8. ตั้งค่า Nginx

/etc/nginx/sites-available/pegasus_erp:

server {
    listen 80;
    server_name erp.example.com;

    root /var/www/pegasus_erp/public;
    index index.php index.html;

    client_max_body_size 32M;

    location / {
        try_files $uri $uri/ /index.php?$query_string;
    }

    location ~ \.php$ {
        include snippets/fastcgi-php.conf;
        fastcgi_pass unix:/var/run/php/php8.2-fpm.sock;
        fastcgi_param SCRIPT_FILENAME $realpath_root$fastcgi_script_name;
    }

    location ~ /\.(env|git) { deny all; }
    location ~* \.(log|sql|dump)$ { deny all; }
}
sudo ln -s /etc/nginx/sites-available/pegasus_erp /etc/nginx/sites-enabled/
sudo nginx -t && sudo systemctl reload nginx

9. เปิด HTTPS (Let's Encrypt)

sudo certbot --nginx -d erp.example.com
sudo certbot renew --dry-run

10. ปรับแต่ง php-fpm

/etc/php/8.2/fpm/php.ini:

memory_limit = 256M
upload_max_filesize = 32M
post_max_size = 32M
date.timezone = Asia/Bangkok
session.gc_maxlifetime = 7200
sudo systemctl restart php8.2-fpm

11. Auto Backup (cron)

sudo crontab -e

# สำรองทุกวันตอน 3:00 (เก็บ 30 วัน)
0 3 * * * PGPASSWORD='STRONG_PASSWORD_HERE' pg_dump -U pegasus_user -h localhost \
  -d pegasus_erp -Fc -f /var/www/pegasus_erp/backups/pegasus_$(date +\%Y\%m\%d).dump && \
  find /var/www/pegasus_erp/backups -name 'pegasus_*.dump' -mtime +30 -delete
