3. ขั้นตอนการติดตั้ง

3.1 อัปเดตระบบ + ติดตั้งแพ็กเกจ

sudo apt update && sudo apt upgrade -y
sudo apt install -y \
    nginx \
    php8.2-fpm php8.2-pgsql php8.2-mbstring php8.2-gd \
    php8.2-xml php8.2-curl php8.2-zip php8.2-intl \
    postgresql-15 postgresql-client-15 postgresql-contrib-15 \
    unzip git curl jq certbot python3-certbot-nginx ufw fail2ban

3.2 Timezone & Firewall

sudo timedatectl set-timezone Asia/Bangkok
sudo ufw allow OpenSSH
sudo ufw allow 'Nginx Full'
sudo ufw --force enable

3.3 ตั้งค่า PostgreSQL

sudo -u postgres psql <<EOF
CREATE USER pegasus_user WITH PASSWORD 'StrongPassword2026!';
CREATE DATABASE pegasus_erp OWNER pegasus_user ENCODING 'UTF8'
       LC_COLLATE 'en_US.UTF-8' LC_CTYPE 'en_US.UTF-8' TEMPLATE template0;
GRANT ALL PRIVILEGES ON DATABASE pegasus_erp TO pegasus_user;
EOF

แก้ไข /etc/postgresql/15/main/pg_hba.conf:

local   pegasus_erp   pegasus_user                       md5
host    pegasus_erp   pegasus_user   127.0.0.1/32        md5
sudo systemctl restart postgresql

5. ตั้งค่า Nginx + PHP-FPM

5.1 Nginx Virtual Host

สร้างไฟล์ /etc/nginx/sites-available/pegasus:

server {
    listen 80;
    server_name erp.tomastech.com;

    root /var/www/pegasus_erp/public;
    index index.php index.html;
    client_max_body_size 32M;

    location ~ /\.(env|git|ht) { deny all; return 404; }
    location ~* \.(log|sql|dump|bak|sh)$ { deny all; return 404; }
    location ^~ /backups/ { deny all; return 404; }

    location ~* \.(?:jpg|jpeg|png|webp|gif|ico|css|js|svg|woff2?)$ {
        expires 7d;
        access_log off;
        try_files $uri =404;
    }

    location / {
        try_files $uri $uri/ /index.php?$query_string;
    }

    location ~ \.php$ {
        include snippets/fastcgi-php.conf;
        fastcgi_pass unix:/var/run/php/php8.2-fpm.sock;
        fastcgi_param SCRIPT_FILENAME $realpath_root$fastcgi_script_name;
    }
}
sudo ln -sf /etc/nginx/sites-available/pegasus /etc/nginx/sites-enabled/pegasus
sudo rm -f /etc/nginx/sites-enabled/default
sudo nginx -t && sudo systemctl reload nginx

5.2 PHP-FPM Configuration

สร้างไฟล์ /etc/php/8.2/fpm/conf.d/99-pegasus.ini:

memory_limit = 256M
upload_max_filesize = 32M
post_max_size = 32M
max_execution_time = 120
date.timezone = Asia/Bangkok
session.gc_maxlifetime = 7200
session.cookie_httponly = 1
display_errors = Off
log_errors = On
error_log = /var/log/php_errors.log
sudo touch /var/log/php_errors.log
sudo chown www-data:www-data /var/log/php_errors.log
sudo systemctl restart php8.2-fpm

5.3 HTTPS (Let's Encrypt)

sudo certbot --nginx -d erp.tomastech.com
sudo certbot renew --dry-run

7. การสำรองข้อมูลอัตโนมัติ

สร้างไฟล์ /var/www/pegasus_erp/deploy/backup-local.sh:

#!/bin/bash
set -e
source /var/www/pegasus_erp/.env
TS=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR=/var/www/pegasus_erp/backups
mkdir -p "$BACKUP_DIR"
PGPASSWORD="$DB_PASS" pg_dump \
    -U "$DB_USER" -h "$DB_HOST" -d "$DB_NAME" \
    -Fc -f "$BACKUP_DIR/pegasus_${TS}.dump"
find "$BACKUP_DIR" -name 'pegasus_*.dump' -mtime +14 -delete
echo "[$(date)] Backup: pegasus_${TS}.dump"
sudo chmod +x /var/www/pegasus_erp/deploy/backup-local.sh
sudo crontab -e

เพิ่มบรรทัด (รันทุกวันตอน 3:00):

0 3 * * * /var/www/pegasus_erp/deploy/backup-local.sh >> /var/log/pegasus_backup.log 2>&1

8. การแก้ปัญหา

อาการวิธีแก้
502 Bad Gatewaysudo systemctl restart php8.2-fpm nginx
ต่อฐานข้อมูลไม่ได้ตรวจสอบ .env / firewall / pg_hba.conf
Session หมดอายุเร็วเพิ่ม session.gc_maxlifetime ใน php.ini
อัปโหลดไฟล์ไม่ได้ตรวจสอบ upload_max_filesize / ownership ของ uploads
ภาษาไทย/ญี่ปุ่นเพี้ยนDB encoding ต้องเป็น UTF-8 + mbstring เปิดใช้งาน
SSL ต่ออายุไม่ได้ตรวจสอบ port 80 เปิด + sudo certbot renew
